class Client:
    def __init__(self, addr):
        self.sock = nnpy.Socket(nnpy.AF_SP, nnpy.REQ)
        self.sock.connect(addr)
    def _send(self, msg):
        self.sock.send(msg)
        return self.sock.recv()
